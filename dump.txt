  private provider: ethers.BrowserProvider | null = null;
  private userAddress: string = '';

  async initialize(address: string): Promise<boolean> {
    if (!window.ethereum) return false;
    try {
      this.userAddress = address;
      this.provider = new ethers.BrowserProvider(window.ethereum);
      console.log(`âœ… RealSwapExecutor inicializado`);
      return true;
    } catch {
      return false;
    }

    if (!this.provider) return 0;
    try {
      const tokenAddress = token === 'USDC' ? ARC_TESTNET.usdc : ARC_TESTNET.eurc;
      const contract = new ethers.Contract(tokenAddress, ERC20_ABI, this.provider);
      const balance = await contract.balanceOf(this.userAddress);
      const decimals = await contract.decimals();
      return parseFloat(ethers.formatUnits(balance, decimals));
    } catch {
      return 0;
    }
  }

export interface RealTradeResult {

class RealAutomatedTrader {
  private isRunning: boolean = false;
  private intervalId: NodeJS.Timeout | null = null;
  private tradeHistory: RealTradeResult[] = [];
  private totalProfit: number = 0;
  private userAddress: string = '';
  private currentNetwork: string = 'arc';
  private useRealSwaps: boolean = true;

    this.userAddress = userAddress;
    this.currentNetwork = network;
    realSwap.initialize(userAddress);
    console.log(`Real Automated Trader iniciado na rede ${network}`);
  }

  async getRealBalance(): Promise<number> {
    return await realSwap.getBalance('USDC');
  }

  async runTradingCycle(tradeAmount: number = 10): Promise<RealTradeResult> {
    const currentBalance = await this.getRealBalance();
    
    if (currentBalance < tradeAmount) {
      return {
        success: false,
        profit: 0,
        message: `Saldo insuficiente: ${currentBalance.toFixed(2)} < ${tradeAmount}`,
        timestamp: Date.now()
      };
    }
    
    // DecisÃ£o do robÃ´
    const actions = ['BUY', 'SELL', 'HOLD'];
    const randomAction = actions[Math.floor(Math.random() * actions.length)];
    
    if (randomAction === 'HOLD') {
      return {
        success: true,
        profit: 0,
        message: `HOLD - Nenhum trade`,
        timestamp: Date.now()
      };
    }
    
    // Abrir LI.FI para swap real
    console.log(`\nðŸš€ ABRINDO LI.FI PARA SWAP REAL!`);
    console.log(`   AÃ§Ã£o: ${randomAction} | Valor: $${tradeAmount}`);
    
    const result = realSwap.openLiFiSwap(randomAction as 'BUY' | 'SELL', tradeAmount);
    
    // Registrar tentativa (lucro serÃ¡ real apÃ³s confirmaÃ§Ã£o manual)
    const profit = 0;
    this.totalProfit += profit;
    
    return {
      success: true,
      profit: profit,
      txHash: result.txHash,
      explorerUrl: result.explorerUrl,
      message: `${randomAction}: ${result.message}`,
      timestamp: Date.now()
    };
  }

  startAutomatedTrading(intervalSeconds: number = 30, tradeAmount: number = 10) {
    if (this.isRunning) return;
    this.isRunning = true;
    console.log(`\nðŸš€ REAL AUTOMATED TRADER INICIADO!`);
    console.log(`   Trade: $${tradeAmount} | Intervalo: ${intervalSeconds}s`);
    
    this.runTradingCycle(tradeAmount);
    
    this.intervalId = setInterval(async () => {
      if (!this.isRunning) return;
      await this.runTradingCycle(tradeAmount);
    }, intervalSeconds * 1000);
  }

  stopAutomatedTrading() {
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = null;
    }
    this.isRunning = false;
    console.log("\nâ¹ï¸ Trading parado");
  }

  getStats() {
    const totalTrades = this.tradeHistory.length;
    const winningTrades = this.tradeHistory.filter(t => t.profit > 0).length;
    const winRate = totalTrades > 0 ? (winningTrades / totalTrades) * 100 : 0;
    
    return {
      totalTrades,
      winningTrades,
      losingTrades: totalTrades - winningTrades,
      winRate: winRate.toFixed(2),
      totalProfit: this.totalProfit.toFixed(4),
      avgProfit: totalTrades > 0 ? (this.totalProfit / totalTrades).toFixed(4) : '0',
      isRunning: this.isRunning
    };
  }

  getHistory() {
    return this.tradeHistory;
  }
}
